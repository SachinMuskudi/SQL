/* SHOW THE TABLE */
SELECT *FROM EMPLOYEE

SELECT *FROM STUDENT

/* INNER JOIN */

SELECT *FROM EMPLOYEE INNER JOIN STUDENT ON EMPLOYEE.ID=STUDENT.ID

/* LEFT OUTER JOIN */
SELECT *FROM EMPLOYEE LEFT OUTER JOIN STUDENT ON EMPLOYEE.ID=STUDENT.ID

/* RIGHT OUTER JOIN */
SELECT *FROM EMPLOYEE RIGHT OUTER JOIN STUDENT ON EMPLOYEE.ID=STUDENT.ID

/* FULL OUTER JOIN */
SELECT *FROM EMPLOYEE FULL OUTER JOIN STUDENT ON EMPLOYEE.ID=STUDENT.ID

/* CROSS JOIN */
SELECT *FROM EMPLOYEE CROSS JOIN STUDENT

/* EQUI JOIN */
SELECT *FROM EMPLOYEE,STUDENT WHERE EMPLOYEE.ID=STUDENT.ID

/* NON-EQUI JOIN */
SELECT *FROM EMPLOYEE,STUDENT WHERE EMPLOYEE.ID > STUDENT.ID
